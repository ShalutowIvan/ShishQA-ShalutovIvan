#/!bin/bash # определяет через какую оболочку будет работать скрипт

options=(
	'1'
	"2"
	'3'
	'4'
	'5'
) # options это название массива для опций, ему присвоены значения, первое значние под номером 0
echo 'Please choose an option' # выводит на экран надпись которая в кавычках
select input in "${options[@]}"; do
	case "$input" in
		"${options[0]}")
			ls
			;;    # выводит список файлов и папок в текущей директории где находится сам файл скрипта
		"${options[1]}")
			date $'+Time: %T\nDate: %D'
			;; выводит текущие дату и время
		"${options[2]}")
			read -p 'What file do you wish to check? ' finput # спрашивает какой файл мы хотим проверить и нужно ввести имя файла, то есть считывает имя файла
			if [[ -d $finput ]]; then # если мы ввели имя папки, то напишет: "название папки которую мы ввели" это директория
				format='%s is a directory.\n'
			elif [[ -f $finput ]]; then 
				format='%s is a file.\n' # иначе если, второе условие - если мы введем название файла, то напишет: "название файла который мы ввели" это файл
			else # 
				format='%s does not exist.\n' # иначе, если ввести что не корректное, то есть не название файла, а что попало, то напишет: "текст который мы ввели" не существует
			fi # конец условия
			printf "$format" "$finput" # формат ввода вывода
			;;
		"${options[3]}")
			read -p 'Please enter filename: ' binput # просит ввести имя файла, считывает имя файла
			cp "$binput"{,.bak} # копирует файл и дописывает ему расширение .bak
			;;
		"${options[4]}")
			break # выходит из скрипта
			;;
		*)
			echo 'Not a valid option.' # прописывает эту надпись если ввести номер опции которой нет в списке.
			;;
	esac
done
